区间模式 SQL
SELECT sb.ts_code, sb.name, MIN(sid.trade_date) AS first_hit_date, COUNT(*) AS hit_days_count FROM stock_indicator_daily sid LEFT JOIN stock_indicator_daily prev ON prev.stock_code = sid.stock_code AND prev.trade_date = (SELECT MAX(trade_date) ...) JOIN stock_basic sb ON sb.ts_code = sid.stock_code WHERE sid.trade_date BETWEEN '20240101' AND '20240115' AND (sid.ma_5 > sid.ma_20 AND prev.ma_5 <= prev.ma_20) AND (sid.volume > (sid.vol_ma_20 * 1.5)) AND sb.name NOT LIKE '%ST%' GROUP BY sb.ts_code, sb.name ORDER BY first_hit_date DESC, hit_days_count DESC LIMIT 200;
